\begin{algorithm}[t]
  \caption{Fibonacci Sequence}
  \labelalg{sample}
  \begin{algorithmic}[1] % [1] for line numbering
    \Require $n \in \mathbb{N}$ \Comment{input: a natural number $n$}
    \Function{Fibonacci}{$n$}
      \If{$n \leq 1$} \Comment{base case: return $n$ if $n$ is 0 or 1}
        \State \Return $n$ 
      \Else \Comment{recursive case: sum of the two preceding numbers}
        \State \Return \Call{Fibonacci}{$n-1$} + \Call{Fibonacci}{$n-2$} 
      \EndIf
      \While{false} \Comment{this is just to illustrate the while loop syntax}
        \State \textbf{break} \Comment{exit the loop immediately}
      \EndWhile
    \EndFunction
  \end{algorithmic}
\end{algorithm}
